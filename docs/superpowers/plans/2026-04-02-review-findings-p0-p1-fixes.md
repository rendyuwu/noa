async function deleteUser() {
  if (!selectedUser) {
    return;
  }

  setConfirmDeleteOpen(false);
  setDeleting(true);
  setActionError(null);
  setActionMessage(null);

  try {
    const response = await fetchWithAuth(`/admin/users/${selectedUser.id}`, {
      method: "DELETE",
    });

    setUsers((current) => current.filter((user) => user.id !== selectedUser.id));

  }
}

- [ ] **Step 3: Fix as never casts in assistant-transport-converter.ts**

Lines 148 and 158 have `content as never` to satisfy the `ThreadMessage` type's strict content typing. The `ThreadMessage` type expects specific content part union arrays. The cleanest fix is to use a type assertion helper that preserves the intent:

```ts
// The content arrays are built dynamically from Record<string, unknown>[] but
// ThreadMessage expects typed content part arrays. Since we build these
// carefully via partsToContent(), a single type assertion at the boundary
// is acceptable — but use the target type instead of `never`.

// Line 144-151: Replace the user return:
if (role === "user") {
  return {
    id,
    createdAt,
    role: "user" as const,
    content,
    attachments: [],
    metadata: { custom: {} },
  } as ThreadMessage;
}

// Line 153-167: Replace the assistant return:
return {
  id,
  createdAt,
  role: "assistant" as const,
  content,
  status: { type: "complete", reason: "stop" },
  metadata: {
    unstable_state: null,
    unstable_annotations: [],
    unstable_data: [],
    steps: [],
    custom: {},
  },
} as ThreadMessage;
```

The key change: remove `content as never` and just use `content` directly since the whole object is already cast `as ThreadMessage` at the end. The `as ThreadMessage` at the object level is a single, intentional boundary cast — cleaner than `as never` on individual properties.
